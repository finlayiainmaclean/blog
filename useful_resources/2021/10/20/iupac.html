class Resolver:
  def __init__(self):
    self.url = "https://cactus.nci.nih.gov/chemical/structure/{0}/{1}"

  def _smiles_to_iupac_cactus(self, smiles):
    try:
      response = requests.get(self.url.format(smiles, 'iupac_name'))
      response.raise_for_status()
      return response.text
    except requests.HTTPError as exception:
      return None


  def _smiles_to_iupac_pubchem(self, smiles):
    try:
      compounds = pubchempy.get_compounds(smiles, namespace='smiles')
      match = compounds[0]
      return match.iupac_name
    except pubchempy.BadRequestError as exception:
      return None

  def to_iupac(self, mols, method='cactus'):
    if type(mols)!=list:
      mols = [mols]
  

    if type(mols[0]) == rdkit.Chem.rdchem.Mol:
      smiles = [Chem.MolToSmiles(mol) for mol in mols]
    elif type(mols[0]) == str:
      smiles = mols
    else:
      raise NotImplementedError('You donut - Input needs to be string or rdkit mol')
    if method=='cactus':
      func = self._smiles_to_iupac_cactus
    elif method=='pubchem':
      func = self._smiles_to_iupac_pubchem
    else:
      raise NotImplementedError('Muppet')
      
    return [func(smile) for smile in tqdm(smiles)]

def prep_sidechain(smi, atom_map_no=1):
    mol = Chem.MolFromSmiles(smi)
    rw_mol = Chem.RWMol(mol)
    remove_idx = -1
    for atm in rw_mol.GetAtoms():
        # Find dummy atom (atomic number 0) and delete
        if atm.GetAtomicNum() == 0:

    return Chem.MolToSmiles(rw_mol)
